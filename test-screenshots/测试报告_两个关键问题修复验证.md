#### 代码实现验证：
通过检查 `app/services/recipe_generator.py` 确认：

```python
# 优先尝试使用 CodingPlan API 动态生成
generation_source = GENERATION_SOURCE_CODINGPLAN  # "codingplan"

if is_api_available():
    # 调用火山引擎API
    result = await volcengine_client.generate_json(...)
    if result:
        result["generation_source"] = GENERATION_SOURCE_CODECINGPLAN
        return result
    else:
        logger.warning("[CodingPlan] API调用失败，将降级使用到内置算法")

# 第二步：降级到内置算法
generation_source = GENERATION_SOURCE_ALGORITHM  # "algorithm"
logger.info(f"[Algorithm Day {day}] 使用内置智能算法生成")
result = generate_day_meals(baby_profile, exclude_names, day)
result["generation_source"] = GENERATION_SOURCE_ALGORITHM
return result
```
